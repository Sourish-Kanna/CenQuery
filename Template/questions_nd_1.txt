SELECT ((SUM(e.main_working_person) + SUM(e.marginal_worker_person)) * 100.0 / NULLIF(SUM(e.total_person), 0)) FROM education_stats e JOIN regions r ON e.state = r.state JOIN tru t ON e.tru_id = t.id WHERE t.name = 'Total' AND r.state = (SELECT state FROM population_stats WHERE tru_id = (SELECT id FROM tru WHERE name = 'Total') ORDER BY persons ASC LIMIT 1);
SELECT SUM(non_working_female) FROM education_stats WHERE tru_id = (SELECT id FROM tru WHERE name = 'Total');
SELECT SUM(non_working_male) FROM education_stats WHERE tru_id = (SELECT id FROM tru WHERE name = 'Total');
SELECT r.area_name, SUM(e.literates_female) AS rural_lit_women FROM education_stats e JOIN regions r ON e.state = r.state JOIN tru t ON e.tru_id = t.id WHERE t.name = 'Rural' GROUP BY r.area_name ORDER BY rural_lit_women DESC LIMIT 1;
SELECT r.area_name, SUM(e.literates_male) AS urban_lit_men FROM education_stats e JOIN regions r ON e.state = r.state JOIN tru t ON e.tru_id = t.id WHERE t.name = 'Urban' GROUP BY r.area_name ORDER BY urban_lit_men DESC LIMIT 1;
SELECT SUM(e.main_working_person) FROM education_stats e JOIN regions r ON e.state = r.state JOIN tru t ON e.tru_id = t.id WHERE r.area_name IN ('Tamil Nadu', 'Kerala', 'Karnataka', 'Andhra Pradesh', 'Telangana') AND t.name = 'Total';
SELECT SUM(e.marginal_worker_person) FROM education_stats e JOIN regions r ON e.state = r.state JOIN tru t ON e.tru_id = t.id WHERE r.area_name IN ('Punjab', 'Haryana', 'Uttar Pradesh', 'Rajasthan') AND t.name = 'Total';
